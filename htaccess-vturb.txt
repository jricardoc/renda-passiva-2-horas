# =========================================================================
# CSP Headers para Vturb Smartplayer
# =========================================================================
# Adicione este arquivo na raiz do seu site WordPress
# Funciona com hospedagens Apache (Hostinger, etc.)

<IfModule mod_headers.c>
    # Content Security Policy para permitir Vturb
    Header set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval' https://cdn.converteai.net https://smartplayer.converteai.net https://scripts.converteai.net https://*.facebook.net https://*.google-analytics.com https://*.googletagmanager.com; connect-src 'self' https://cdn.converteai.net https://api.converteai.net https://*.facebook.com https://*.google-analytics.com; media-src 'self' https://cdn.converteai.net blob: data:; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com data:; img-src 'self' data: https: blob:; frame-src 'self' https://*.facebook.com https://*.google.com;"
</IfModule>

# Cross-Origin headers para m√≠dia HLS
<IfModule mod_headers.c>
    <FilesMatch "\.(m3u8|ts|mp4|webm)$">
        Header set Access-Control-Allow-Origin "*"
        Header set Access-Control-Allow-Methods "GET, OPTIONS"
        Header set Access-Control-Allow-Headers "Origin, X-Requested-With, Content-Type, Accept"
    </FilesMatch>
</IfModule>
